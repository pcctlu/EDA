import pandas as pd
import re
from collections import Counter
import matplotlib.pyplot as plt

# Sample text data
texts = [
    "Natural language processing is a fascinating field.",
    "I love working with text data and NLP models!",
    "Text preprocessing is essential for machine learning tasks.",
    "Understanding word frequencies helps in feature engineering.",
    "NLP involves tokenization, stemming, lemmatization, and more.",
    "Bag of words and n-grams are common text representation methods.",
    "Sentence length and word count can reveal useful patterns.",
    "Exploratory data analysis for text is crucial in data science."
]


df = pd.DataFrame({'text': texts})

# 1. Count characters and words
df['char_count'] = df['text'].apply(len)
df['word_count'] = df['text'].apply(lambda x: len(re.findall(r'\b\w+\b', x)))

# 2. Word frequency analysis (lowercase, alphanumeric only)
all_words = []  # Create an empty list to hold all words

# Loop through each text, convert to lowercase, extract only words
for text in df['text']:
    words = re.findall(r'\b\w+\b', text.lower())  # Extract all lowercase words
    all_words.extend(words)  # Add to list


# Count how frequently each word appears
word_freq = Counter(all_words)

# Extract top 10 most common words and their counts
most_common_words = word_freq.most_common(10)

# 3. Sentence length analysis (in words)
df['sentence_word_len'] = df['word_count']  # each row is one sentence

# 4. Basic Bag-of-Words & n-grams (bigrams) length analysis
def get_ngrams(tokens, n):
    return [' '.join(tokens[i:i+n]) for i in range(len(tokens)-n+1)]


df['tokens'] = df['text'].apply(lambda x: re.findall(r'\b\w+\b', x.lower()))

df['bigrams'] = df['tokens'].apply(lambda x: get_ngrams(x, 2))

df['bigram_count'] = df['bigrams'].apply(len)

# Output results

results = {
    'char_counts': df['char_count'].tolist(),
    'word_counts': df['word_count'].tolist(),
    'most_common_words': most_common_words,
    'sentence_lengths': df['sentence_word_len'].tolist(),
    'avg_word_count': df['word_count'].mean(),
    'avg_char_count': df['char_count'].mean(),
    'bigram_counts': df['bigram_count'].tolist()
}

results
